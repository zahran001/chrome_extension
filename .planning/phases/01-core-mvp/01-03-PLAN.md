**AABB (src/extraction/aabb.ts):**
- `rectsIntersect(rect1: Rect, rect2: Rect): boolean`
  - Overlapping → true
  - Touching edge → true (inclusive)
  - Adjacent (gap of 0) → true
  - Non-overlapping with gap → false
  - Zero-size rect (width=0 or height=0) → false (fast-fail before AABB math)
  - Negative coordinates → handled correctly (rect3: {left:-100, right:-10, top:0, bottom:10})
  - One fully inside other → true

- `pointInRect(x: number, y: number, rect: Rect): boolean`
  - Point inside → true
  - Point on boundary → true
  - Point outside → false

**Visibility (src/extraction/visibility.ts):**
- `isVisible(element: Element): boolean`
  - Visible element → true
  - display:none → false
  - visibility:hidden → false
  - opacity:0 → false (parseFloat check)
  - Zero clientWidth → false
  - Zero clientHeight → false

**TreeWalker (src/extraction/tree-walker.ts):**
- `extractVisibleText(root: Element, selectionBounds: DOMRect): string`
  - Returns concatenated text of visible text nodes within selectionBounds
  - Skips display:none nodes (via isVisible)
  - Skips zero-dimension nodes (fast-fail per CONTEXT.md constraint)
  - Uses rectsIntersect for AABB check
  - NEVER uses innerHTML — reads textNode.data only
  - Returns '' when nothing in bounds

export interface Rect {
  left: number;
  right: number;
  top: number;
  bottom: number;
}

export function rectsIntersect(rect1: Rect, rect2: Rect): boolean {
  // Zero-size fast-fail
  if (rect1.right - rect1.left <= 0 || rect1.bottom - rect1.top <= 0) return false;
  if (rect2.right - rect2.left <= 0 || rect2.bottom - rect2.top <= 0) return false;
  // AABB intersection
  return (
    rect1.left <= rect2.right &&
    rect1.right >= rect2.left &&
    rect1.top <= rect2.bottom &&
    rect1.bottom >= rect2.top
  );
}

export function pointInRect(x: number, y: number, rect: Rect): boolean {
  return x >= rect.left && x <= rect.right && y >= rect.top && y <= rect.bottom;
}
```

export function isVisible(element: Element): boolean {
  const style = getComputedStyle(element);
  if (style.display === 'none') return false;
  if (style.visibility === 'hidden') return false;
  if (parseFloat(style.opacity) === 0) return false;
  // Dimension check (after style check to avoid unnecessary layout)
  if (element.clientWidth === 0 || element.clientHeight === 0) return false;
  return true;
}
```
